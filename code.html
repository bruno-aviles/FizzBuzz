                        //Display Results to user
                        function displayResults(){

                            //get user inputs
                            let fNumber = document.getElementById("fizzNumber").value;
                            let bNumber = document.getElementById("buzzNumber").value;
                            
                            verifyNumbers(fNumber,bNumber);
                        }

                        //Verify FIZZBUZZ
                        function verifyNumbers(fizz,buzz){

                            let displayNumbers = document.getElementById("results");
                            // store all <td> elements from for loop
                            let templateRows = "";

                            // verifies to see if user inputs are between 1 to 100
                            if(fizz < 1 || fizz > 100){
                                templateRows += `<tr><td>Enter A Number Between 1 - 100</td></tr>`;
                                return displayNumbers.innerHTML = templateRows;
                            } else if(buzz < 1 || buzz > 100){
                                templateRows += `<tr><td>Enter A Number Between 1 - 100</td></tr>`;
                                return displayNumbers.innerHTML = templateRows;
                            }

                            // loop through input numbers counting forward and add <td> element for each number
                            if(fizz < buzz){
                                for(number=fizz;number <= buzz;number++){
                                    // replace the numbers that are a multiple of 3 and 5 with "FIZZ" and "BUZZ"
                                    if(number % 3 == 0 && number % 5 == 0){
                                        templateRows += `<tr><td>FIZZBUZZ</td></tr>`;
                                    } else if(number % 3 == 0){
                                        templateRows += `<tr><td>FIZZ</td></tr>`;
                                    } else if(number % 5 == 0){
                                        templateRows += `<tr><td>BUZZ</td></tr>`;
                                    } else {
                                        templateRows += `<tr><td>${number}</td></tr>`;
                                    }
                            
                                    // insert all stored values from templateRows to <tbody>
                                    displayNumbers.innerHTML = templateRows;      
                                }
                            } else if(fizz > buzz){
                                for(number=fizz;number >= buzz;number--){
                                    // replace the numbers that are a multiple of 3 and 5 with "FIZZ" and "BUZZ"
                                    if(number % 3 == 0 && number % 5 == 0){
                                        templateRows += `<tr><td>FIZZBUZZ</td></tr>`;
                                    } else if(number % 3 == 0){
                                        templateRows += `<tr><td>FIZZ</td></tr>`;
                                    } else if(number % 5 == 0){
                                        templateRows += `<tr><td>BUZZ</td></tr>`;
                                    } else {
                                        templateRows += `<tr><td>${number}</td></tr>`;
                                    }
                            
                                    // insert all stored values from templateRows to <tbody>
                                    displayNumbers.innerHTML = templateRows;      
                                }
                            }    
                        }

The Code is Structured in two functions

displayResults

displayResults function obtains(gets) the values from the input fields and displays them on a table format. The values are stored in variables (fNumber) and (bNumber). Variables (fNumber) and (bNumber) are passed through function verifyNumbers.

verifyNumbers

verifyNumbers takes two parameters "fizz" and "buzz" and contains two variables (displayNumbers) and (templateRows). Variable (displayNumbers) obtains the element that will display the results. "fizz" and "buzz" values are checked to see if the values are within numbers 1 to 100. If the values are not within the specified number range, a message will display in the results container reassuring the user to input the appropriate values.

A for loop is used to iterate the numbers starting from the "fizz" number to the "buzz" number. For each number that is being iterated, if the number is a multiple of '3', then the word FIZZ will be displayed in the results container. If the number is a multiple of '5', then the word BUZZ will be displayed in the results container. If the number is a multiple of both '3' and '5', then the word FIZZBUZZ will be displayed in the results container. All other numbers that are not a multiple of '3' and '5' will be displayed as numbers.
